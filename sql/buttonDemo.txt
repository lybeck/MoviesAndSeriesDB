<script>
function addField()
{
  var inputs = document.getElementsByTagName('input');
  var nextSearch=inputs.length+1;
  var searchString="\"Search #" + nextSearch + "\"";
  var searchType="\"text\"";
  
  var code="<input type=" + searchType + " value=" +
  searchString + " id=\"" + nextSearch + "\">" + 
  "</div><br><div id=\"asdasd" + (nextSearch+1) + "\"></div>";

  document.getElementById("asdasd" + nextSearch).innerHTML=code;
}
</script>

<script>
function myFunction()
{
 var inputs = document.getElementsByTagName('input');
 for(var i=0; i<inputs.length; i++){
   var txt="Search #" + (i+1);
   document.getElementById((i+1)).value=txt;
 }
}
</script>
