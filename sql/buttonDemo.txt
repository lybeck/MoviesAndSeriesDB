<button onclick="addField()">Add Field</button> &emsp;

<button onclick="myFunction()">Rename Field</button><br><br>
<input type="text" value="#1" id="1"><br>
<div id="asdasd2"></div>


<script>
function addField()
{
  var inputs = document.getElementsByTagName('input');
  var nextSearch=inputs.length+1;
  if(nextSearch>4){
     return;
  }
  var searchString="\"Search #" + nextSearch + "\"";
  var searchType="\"text\"";
  
  var code="<input type=" + searchType + " value=" +
  searchString + " id=\"" + nextSearch + "\">" + 
  "<br><div id=\"asdasd" + (nextSearch+1) + "\"></div>";

  document.getElementById("asdasd" + nextSearch).innerHTML=code;
}
</script>

<script>
function removeField()
{
  var inputs = document.getElementsByTagName('input');
  var lastSearch=inputs.length;
  if(lastSearch == 1){
     return;
  }
  document.getElementById("asdasd" + lastSearch).innerHTML="";
}
</script>

<script>
function myFunction()
{
 var inputs = document.getElementsByTagName('input');
 for(var i=0; i<inputs.length; i++){
   var txt="Search #" + (i+1);
   document.getElementById((i+1)).value=txt;
 }
}
</script>
